import streamlit as st
import numpy as np
import pandas as pd
from tensorflow.keras.models import load_model
from tensorflow.keras.preprocessing import image
import tensorflow as tf
import matplotlib.pyplot as plt

# Parameters
IMG_SIZE = 224
CLASS_NAMES = ["healthy", "non_spiral", "parkinson"]

# Load model once
@st.cache_resource
def load_trained_model():
    return load_model("artifacts/best_model.h5")

model = load_trained_model()

# Page Config
st.set_page_config(page_title="Parkinson's Spiral Detection", page_icon="🌀", layout="wide")

# Custom CSS
st.markdown("""
    <style>
        .main { background-color: #f7f9fc; }
        .stFileUploader { border: 2px dashed #4CAF50; padding: 10px; border-radius: 10px; }
        .prediction { font-size: 24px; font-weight: bold; color: #2c3e50; }
    </style>
""", unsafe_allow_html=True)

# Title
st.title("🌀 Parkinson's Spiral Detection")
st.markdown("Upload a spiral image and let the AI classify it into **Healthy**, **Parkinson**, or **Non-Spiral**.")

# File uploader
uploaded_file = st.file_uploader("📂 Upload Spiral Image", type=["jpg", "jpeg", "png"])

if uploaded_file is not None:
    # Left-Right layout
    col1, col2 = st.columns([1, 2])

    with col1:
        st.image(uploaded_file, caption="Uploaded Spiral", use_column_width=True)

    with col2:
        # Preprocess
        img = image.load_img(uploaded_file, target_size=(IMG_SIZE, IMG_SIZE))
        img_array = image.img_to_array(img) / 255.0
        img_array = np.expand_dims(img_array, axis=0)

        # Prediction
        preds = model.predict(img_array)
        predicted_class = CLASS_NAMES[np.argmax(preds)]
        confidence = np.max(preds)

        # Display prediction
        st.markdown(f"<p class='prediction'>Prediction: {predicted_class} ✅ (Confidence: {confidence:.2f})</p>", unsafe_allow_html=True)

        # Probabilities bar chart
        st.write("### 🔎 Class Probabilities")
        prob_df = pd.DataFrame(preds[0], index=CLASS_NAMES, columns=["Probability"])
        st.bar_chart(prob_df)

# Footer
st.markdown("---")
st.markdown("💡 *Built with TensorFlow + Streamlit* | Author: **Your Name**")
